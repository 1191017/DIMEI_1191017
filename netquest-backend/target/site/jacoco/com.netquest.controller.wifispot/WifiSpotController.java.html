@RequiredArgsConstructor
@RestController
@RequestMapping("api/wifi-spot")
public class WifiSpotController {
    private final WifiSpotService wifiSpotService;

    // 5.3.2 - Update Wi-Fi spot description
    @PutMapping("/mysql/{wifiSpotId}/description")
    public ResponseEntity<String> updateWifiSpotDescriptionMySQL(@PathVariable String wifiSpotId, @RequestBody Map<String, String> payload) {
        String newDescription = payload.get("description");
        wifiSpotService.updateWifiSpotDescription(UUID.fromString(wifiSpotId), newDescription, true, false, false);
        return ResponseEntity.ok("Wi-Fi spot description updated successfully.");
    }

    @PutMapping("/mongodb/{wifiSpotId}/description")
    public ResponseEntity<String> updateWifiSpotDescriptionMongoDB(@PathVariable String wifiSpotId, @RequestBody Map<String, String> payload) {
        String newDescription = payload.get("description");
        wifiSpotService.updateWifiSpotDescription(UUID.fromString(wifiSpotId), newDescription, false, true, false);
        return ResponseEntity.ok("Wi-Fi spot description updated successfully.");
    }

    @PutMapping("/cassandra/{wifiSpotId}/description")
    public ResponseEntity<String> updateWifiSpotDescriptionCassandra(@PathVariable String wifiSpotId, @RequestBody Map<String, String> payload) {
        String newDescription = payload.get("description");
        wifiSpotService.updateWifiSpotDescription(UUID.fromString(wifiSpotId), newDescription, false, false, true);
        return ResponseEntity.ok("Wi-Fi spot description updated successfully.");
    }
}
